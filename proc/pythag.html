    pure real(wp) function pythag(a,b)
    implicit none

    real(wp),intent(in) :: a , b

    real(wp) :: p , q , r , s , t

    p = max(abs(a),abs(b))
    q = min(abs(a),abs(b))

    if ( q==0.0_wp ) then
        pythag = p
    else
        do
            r = (q/p)**2
            t = 4.0_wp + r
            if ( t==4.0_wp ) then
                pythag = p
                exit
            else
                s = r/t
                p = p + 2.0_wp*p*s
                q = q*s
            endif
        end do
    endif

    end function pythag
